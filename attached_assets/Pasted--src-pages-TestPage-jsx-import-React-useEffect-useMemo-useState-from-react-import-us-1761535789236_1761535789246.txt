// src/pages/TestPage.jsx
import React, { useEffect, useMemo, useState } from "react";
import { useNavigate, useParams } from "react-router-dom";
import { 
  ArrowLeft, 
  BookOpen, 
  CheckCircle2, 
  XCircle, 
  AlertCircle, 
  Trophy,
  RefreshCw,
  Home,
  Eye,
  EyeOff
} from "lucide-react";
import logo from "../assets/voplogo1.webp";

const MAX_ATTEMPTS = 4;
const PASS_MARK = 50; // percent

// Helper localStorage keys
const attemptsKey = (courseId, lessonId) => `vop_test_attempts_${courseId}_${lessonId}`;
const passedKey = (courseId, lessonId) => `vop_test_passed_${courseId}_${lessonId}`;
const latestScoreKey = (courseId, lessonId) => `vop_test_latest_score_${courseId}_${lessonId}`;

export default function TestPage() {
  const { courseId = "discover", lessonId = "1" } = useParams();
  const navigate = useNavigate();

  // Dummy questions
  const dummyQuestions = useMemo(
    () => [
      {
        id: "q1",
        type: "mcq",
        text: "What is the primary theme of the lesson?",
        options: ["Faith", "Service", "Love", "Hope"],
        correctIndex: 0,
        explanation: "The lesson emphasizes faith as the central theme.",
      },
      {
        id: "q2",
        type: "tf",
        text: "The lesson teaches that faith alone is sufficient without works.",
        correct: false,
        explanation: "The lesson stresses that faith is important, but it pairs with practice and works.",
      },
      {
        id: "q3",
        type: "mcq",
        text: "Which scripture is highlighted in the lesson?",
        options: ["John 3:16", "Romans 8:28", "Psalms 23:1", "Matthew 5:9"],
        correctIndex: 2,
        explanation: "Psalms 23:1 is used to illustrate God's guiding care in the lesson.",
      },
    ],
    []
  );

  const [questions] = useState(dummyQuestions);
  const [userAnswers, setUserAnswers] = useState({});
  const [submitted, setSubmitted] = useState(false);
  const [score, setScore] = useState(null);
  const [attempts, setAttempts] = useState(() => {
    const val = parseInt(localStorage.getItem(attemptsKey(courseId, lessonId)) || "0", 10);
    return isNaN(val) ? 0 : val;
  });
  const [passed, setPassed] = useState(() => localStorage.getItem(passedKey(courseId, lessonId)) === "true");
  const [showCorrections, setShowCorrections] = useState(false);

  useEffect(() => {
    const sc = localStorage.getItem(latestScoreKey(courseId, lessonId));
    if (sc) setScore(Number(sc));
  }, [courseId, lessonId]);

  useEffect(() => {
    function onStorage(e) {
      if (e.key === attemptsKey(courseId, lessonId)) {
        setAttempts(parseInt(e.newValue || "0", 10) || 0);
      }
      if (e.key === passedKey(courseId, lessonId)) {
        setPassed(e.newValue === "true");
      }
    }
    window.addEventListener("storage", onStorage);
    return () => window.removeEventListener("storage", onStorage);
  }, [courseId, lessonId]);

  function handleAnswerChange(q, value) {
    setUserAnswers((prev) => ({ ...prev, [q.id]: value }));
  }

  function grade() {
    let correct = 0;
    questions.forEach((q) => {
      const ans = userAnswers[q.id];
      if (q.type === "mcq") {
        if (typeof ans === "number" && ans === q.correctIndex) correct++;
      } else if (q.type === "tf") {
        if (typeof ans === "boolean" && ans === !!q.correct) correct++;
      }
    });
    const percent = Math.round((correct / questions.length) * 100);
    return { correct, total: questions.length, percent };
  }

  function handleSubmit(e) {
    e.preventDefault();
    const unanswered = questions.filter((q) => userAnswers[q.id] === undefined || userAnswers[q.id] === null || userAnswers[q.id] === "");
    if (unanswered.length > 0) {
      alert(`Please answer all ${questions.length} questions before submitting.`);
      return;
    }

    const newAttempts = attempts + 1;
    localStorage.setItem(attemptsKey(courseId, lessonId), String(newAttempts));
    setAttempts(newAttempts);

    const result = grade();
    setScore(result.percent);
    localStorage.setItem(latestScoreKey(courseId, lessonId), String(result.percent));

    const didPass = result.percent >= PASS_MARK;
    if (didPass) {
      localStorage.setItem(passedKey(courseId, lessonId), "true");
      setPassed(true);
    }

    setSubmitted(true);
    setShowCorrections(true);
  }

  function handleRetry() {
    if (attempts >= MAX_ATTEMPTS) return;
    setSubmitted(false);
    setShowCorrections(false);
    setUserAnswers({});
  }

  function canRetry() {
    return !passed && attempts < MAX_ATTEMPTS;
  }

  function renderOptionButton(q, idx) {
    const selected = userAnswers[q.id] === idx;
    const isCorrect = idx === q.correctIndex;
    
    let styling = "border-2 border-gray-200 bg-white hover:border-orange-300 hover:bg-orange-50";
    
    if (submitted && showCorrections) {
      if (isCorrect) {
        styling = "border-2 border-green-400 bg-green-50";
      } else if (selected) {
        styling = "border-2 border-red-400 bg-red-50";
      } else {
        styling = "border-2 border-gray-200 bg-gray-50";
      }
    } else if (selected) {
      styling = "border-2 border-orange-500 bg-orange-50";
    }

    return (
      <button
        key={idx}
        type="button"
        onClick={() => !submitted && handleAnswerChange(q, idx)}
        disabled={submitted}
        className={`${styling} px-4 py-3 rounded-xl w-full text-left transition-all duration-200 hover:scale-[1.01] flex items-center gap-3 group`}
      >
        <div className={`w-8 h-8 rounded-lg flex items-center justify-center font-bold text-sm ${
          submitted && showCorrections
            ? isCorrect
              ? "bg-green-500 text-white"
              : selected
              ? "bg-red-500 text-white"
              : "bg-gray-200 text-gray-600"
            : selected
            ? "bg-orange-500 text-white"
            : "bg-gray-200 text-gray-600 group-hover:bg-orange-200 group-hover:text-orange-700"
        }`}>
          {String.fromCharCode(65 + idx)}
        </div>
        <div className="flex-1 text-gray-800 font-medium">{q.options[idx]}</div>
        {submitted && showCorrections && isCorrect && (
          <CheckCircle2 size={20} className="text-green-500" />
        )}
        {submitted && showCorrections && selected && !isCorrect && (
          <XCircle size={20} className="text-red-500" />
        )}
      </button>
    );
  }

  function renderTFButton(q, value) {
    const selected = userAnswers[q.id] === value;
    const isCorrect = value === q.correct;
    
    let styling = "border-2 border-gray-200 bg-white hover:border-orange-300 hover:bg-orange-50";
    
    if (submitted && showCorrections) {
      if (isCorrect) {
        styling = "border-2 border-green-400 bg-green-50";
      } else if (selected) {
        styling = "border-2 border-red-400 bg-red-50";
      } else {
        styling = "border-2 border-gray-200 bg-gray-50";
      }
    } else if (selected) {
      styling = "border-2 border-orange-500 bg-orange-50";
    }

    return (
      <button
        key={String(value)}
        type="button"
        onClick={() => !submitted && handleAnswerChange(q, value)}
        disabled={submitted}
        className={`${styling} px-4 py-3 rounded-xl w-full text-left transition-all duration-200 hover:scale-[1.01] flex items-center gap-3 group`}
      >
        <div className={`w-8 h-8 rounded-lg flex items-center justify-center font-bold text-sm ${
          submitted && showCorrections
            ? isCorrect
              ? "bg-green-500 text-white"
              : selected
              ? "bg-red-500 text-white"
              : "bg-gray-200 text-gray-600"
            : selected
            ? "bg-orange-500 text-white"
            : "bg-gray-200 text-gray-600 group-hover:bg-orange-200 group-hover:text-orange-700"
        }`}>
          {value ? "T" : "F"}
        </div>
        <div className="flex-1 text-gray-800 font-medium">{value ? "True" : "False"}</div>
        {submitted && showCorrections && isCorrect && (
          <CheckCircle2 size={20} className="text-green-500" />
        )}
        {submitted && showCorrections && selected && !isCorrect && (
          <XCircle size={20} className="text-red-500" />
        )}
      </button>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-orange-50 via-amber-50 to-white relative overflow-hidden">
      {/* Animated background orbs */}
      <div className="absolute inset-0 overflow-hidden pointer-events-none">
        <div className="absolute top-0 left-0 w-96 h-96 bg-orange-200 rounded-full mix-blend-multiply filter blur-3xl opacity-20 animate-blob"></div>
        <div className="absolute top-0 right-0 w-96 h-96 bg-amber-200 rounded-full mix-blend-multiply filter blur-3xl opacity-20 animate-blob animation-delay-2000"></div>
        <div className="absolute bottom-0 left-1/2 w-96 h-96 bg-orange-300 rounded-full mix-blend-multiply filter blur-3xl opacity-20 animate-blob animation-delay-4000"></div>
      </div>

      {/* Header */}
      <header className="relative z-10 bg-white/80 backdrop-blur-lg shadow-lg border-b border-orange-100">
        <div className="flex items-center justify-between p-4 md:px-12">
          <div className="flex items-center gap-3">
            <div className="relative group">
              <div className="absolute -inset-2 bg-gradient-to-r from-orange-400 to-amber-400 rounded-full blur opacity-25 group-hover:opacity-40 transition duration-500"></div>
              <img src={logo} alt="VOP" className="relative h-12 w-auto rounded-md drop-shadow" />
            </div>
            <div>
              <div className="text-xl font-bold bg-gradient-to-r from-orange-600 to-amber-600 bg-clip-text text-transparent">
                VOP Virtual School
              </div>
              <div className="text-xs text-gray-600 font-medium">Lesson Assessment</div>
            </div>
          </div>

          <div className="flex items-center gap-3">
            <button
              onClick={() => navigate(-1)}
              className="flex items-center gap-2 px-4 py-2 rounded-xl bg-gray-100 text-gray-700 hover:bg-gray-200 transition-all font-semibold"
            >
              <ArrowLeft size={18} />
              <span className="hidden md:inline">Back</span>
            </button>
            <button
              onClick={() => navigate(`/dashboard/learner`)}
              className="flex items-center gap-2 px-5 py-2 rounded-xl bg-gradient-to-r from-orange-500 to-amber-500 text-white hover:from-orange-600 hover:to-amber-600 transition-all shadow-lg hover:shadow-xl font-semibold"
            >
              <Home size={18} />
              <span className="hidden md:inline">Dashboard</span>
            </button>
          </div>
        </div>
      </header>

      {/* Main Content */}
      <main className="relative z-10 p-6 md:p-12 max-w-5xl mx-auto">
        <div className="bg-white/90 backdrop-blur-xl rounded-3xl shadow-2xl border border-orange-100 overflow-hidden">
          {/* Test Header */}
          <div className="bg-gradient-to-r from-orange-500 to-amber-500 p-8 text-white">
            <div className="flex items-start justify-between">
              <div>
                <div className="flex items-center gap-2 mb-2">
                  <BookOpen size={24} />
                  <h1 className="text-3xl font-bold">Lesson {lessonId} Assessment</h1>
                </div>
                <p className="text-orange-100">
                  Answer all questions to test your understanding of the lesson
                </p>
              </div>
              <div className="bg-white/20 backdrop-blur-sm rounded-xl px-6 py-4 text-center">
                <div className="text-sm text-orange-100 mb-1">Attempts</div>
                <div className="text-3xl font-bold">{attempts}/{MAX_ATTEMPTS}</div>
                <div className="text-xs text-orange-100 mt-1">Pass: {PASS_MARK}%</div>
              </div>
            </div>
          </div>

          {/* Questions Form */}
          <form onSubmit={handleSubmit} className="p-8 md:p-10">
            <div className="space-y-6">
              {questions.map((q, idx) => {
                const answered = userAnswers[q.id] !== undefined && userAnswers[q.id] !== null && userAnswers[q.id] !== "";
                const isCorrectAnswer = submitted && showCorrections && (
                  (q.type === "mcq" && userAnswers[q.id] === q.correctIndex) ||
                  (q.type === "tf" && userAnswers[q.id] === q.correct)
                );

                return (
                  <div 
                    key={q.id} 
                    className={`p-6 rounded-2xl border-2 transition-all ${
                      submitted && showCorrections
                        ? isCorrectAnswer
                          ? "bg-green-50 border-green-200"
                          : "bg-red-50 border-red-200"
                        : answered
                        ? "bg-orange-50 border-orange-200"
                        : "bg-white border-gray-200"
                    }`}
                  >
                    <div className="flex items-start justify-between mb-4">
                      <div className="flex items-start gap-3 flex-1">
                        <div className={`w-10 h-10 rounded-xl flex items-center justify-center font-bold text-lg shrink-0 ${
                          submitted && showCorrections
                            ? isCorrectAnswer
                              ? "bg-green-500 text-white"
                              : "bg-red-500 text-white"
                            : "bg-gradient-to-r from-orange-500 to-amber-500 text-white"
                        }`}>
                          {idx + 1}
                        </div>
                        <div className="flex-1">
                          <div className="text-lg font-semibold text-gray-800 leading-relaxed">
                            {q.text}
                          </div>
                        </div>
                      </div>
                      <div className={`px-3 py-1 rounded-full text-xs font-semibold ${
                        q.type === "mcq" 
                          ? "bg-blue-100 text-blue-700" 
                          : "bg-purple-100 text-purple-700"
                      }`}>
                        {q.type === "mcq" ? "Multiple Choice" : "True / False"}
                      </div>
                    </div>

                    <div className="space-y-3">
                      {q.type === "mcq" && q.options.map((opt, i) => renderOptionButton(q, i))}
                      {q.type === "tf" && (
                        <div className="grid grid-cols-2 gap-3">
                          {renderTFButton(q, true)}
                          {renderTFButton(q, false)}
                        </div>
                      )}
                    </div>

                    {/* Explanation */}
                    {submitted && showCorrections && (
                      <div className="mt-4 p-4 bg-white rounded-xl border-2 border-gray-200">
                        <div className="flex items-start gap-2">
                          <AlertCircle size={20} className="text-orange-500 mt-0.5 shrink-0" />
                          <div>
                            <div className="font-semibold text-gray-800 mb-1">
                              Correct Answer: {q.type === "mcq" ? q.options[q.correctIndex] : q.correct ? "True" : "False"}
                            </div>
                            <div className="text-sm text-gray-600">{q.explanation}</div>
                          </div>
                        </div>
                      </div>
                    )}
                  </div>
                );
              })}
            </div>

            {/* Action Bar */}
            <div className="mt-8 flex flex-col sm:flex-row items-center justify-between gap-4 pt-6 border-t-2 border-gray-200">
              <div className="flex items-center gap-3">
                {submitted && passed && (
                  <div className="flex items-center gap-2 px-4 py-2 bg-green-100 border-2 border-green-300 rounded-xl text-green-700 font-semibold">
                    <Trophy size={20} />
                    Passed!
                  </div>
                )}
                {submitted && !passed && (
                  <div className="flex items-center gap-2 px-4 py-2 bg-red-100 border-2 border-red-300 rounded-xl text-red-700 font-semibold">
                    <XCircle size={20} />
                    Not Passed
                  </div>
                )}
              </div>

              <div className="flex items-center gap-3">
                {!submitted ? (
                  <button
                    type="submit"
                    className="flex items-center gap-2 px-8 py-3 rounded-xl bg-gradient-to-r from-orange-500 to-amber-500 text-white hover:from-orange-600 hover:to-amber-600 transition-all duration-300 shadow-lg hover:shadow-xl transform hover:scale-[1.02] font-semibold text-lg"
                  >
                    <CheckCircle2 size={20} />
                    Submit Test
                  </button>
                ) : (
                  <>
                    {score !== null && (
                      <div className="text-center px-6 py-3 bg-gradient-to-r from-orange-50 to-amber-50 rounded-xl border-2 border-orange-200">
                        <div className="text-sm text-gray-600 font-medium">Your Score</div>
                        <div className={`text-3xl font-bold ${
                          score >= PASS_MARK ? "text-green-600" : "text-red-600"
                        }`}>
                          {score}%
                        </div>
                      </div>
                    )}
                    
                    <button
                      type="button"
                      onClick={() => setShowCorrections((s) => !s)}
                      className="flex items-center gap-2 px-6 py-3 rounded-xl bg-gray-100 text-gray-700 hover:bg-gray-200 transition-all font-semibold"
                    >
                      {showCorrections ? <EyeOff size={20} /> : <Eye size={20} />}
                      {showCorrections ? "Hide" : "Show"} Answers
                    </button>

                    {canRetry() && (
                      <button
                        type="button"
                        onClick={handleRetry}
                        className="flex items-center gap-2 px-6 py-3 rounded-xl bg-gradient-to-r from-orange-500 to-amber-500 text-white hover:from-orange-600 hover:to-amber-600 transition-all shadow-lg hover:shadow-xl font-semibold"
                      >
                        <RefreshCw size={20} />
                        Retry ({MAX_ATTEMPTS - attempts} left)
                      </button>
                    )}
                  </>
                )}
              </div>
            </div>
          </form>

          {/* Results Section */}
          {submitted && (
            <div className={`p-8 border-t-2 ${
              passed ? "bg-gradient-to-r from-green-50 to-emerald-50 border-green-200" : "bg-gradient-to-r from-red-50 to-rose-50 border-red-200"
            }`}>
              <div className="flex items-center justify-between">
                <div className="flex items-start gap-4">
                  <div className={`w-16 h-16 rounded-2xl flex items-center justify-center ${
                    passed ? "bg-green-500" : "bg-red-500"
                  }`}>
                    {passed ? <Trophy className="text-white" size={32} /> : <XCircle className="text-white" size={32} />}
                  </div>
                  <div>
                    <h3 className={`text-2xl font-bold mb-2 ${passed ? "text-green-700" : "text-red-700"}`}>
                      {passed ? "Congratulations!" : "Keep Trying!"}
                    </h3>
                    <p className="text-gray-700">
                      {passed 
                        ? "You've successfully passed this assessment. Continue to the next lesson or return to the course."
                        : `You've used ${attempts} of ${MAX_ATTEMPTS} attempts. ${canRetry() ? "Review the answers and try again." : "No more attempts available."}`
                      }
                    </p>
                  </div>
                </div>
                <div className="flex gap-3">
                  <button
                    onClick={() => navigate(`/courses/${courseId}`)}
                    className="px-6 py-3 rounded-xl bg-white border-2 border-gray-300 text-gray-700 hover:bg-gray-50 transition-all font-semibold"
                  >
                    Back to Course
                  </button>
                  {passed && (
                    <button
                      onClick={() => navigate(`/dashboard/learner`)}
                      className="px-6 py-3 rounded-xl bg-gradient-to-r from-orange-500 to-amber-500 text-white hover:from-orange-600 hover:to-amber-600 transition-all shadow-lg hover:shadow-xl font-semibold"
                    >
                      Continue Learning
                    </button>
                  )}
                </div>
              </div>
            </div>
          )}
        </div>
      </main>

      <style>{`
        @keyframes blob {
          0%, 100% { transform: translate(0px, 0px) scale(1); }
          33% { transform: translate(30px, -50px) scale(1.1); }
          66% { transform: translate(-20px, 20px) scale(0.9); }
        }
        .animate-blob {
          animation: blob 7s infinite;
        }
        .animation-delay-2000 {
          animation-delay: 2s;
        }
        .animation-delay-4000 {
          animation-delay: 4s;
        }
      `}</style>
    </div>
  );
}
